String mergeSql = 
    "MERGE INTO DEHTGEMO.gem_patrimonio t " +
    "USING (SELECT ? AS fec_periodo, ? AS patrimonio, ? AS estado, ? AS fecha_mod FROM dual) src " +
    "ON (t.fec_periodo = src.fec_periodo) " +
    "WHEN MATCHED THEN " +
    "  UPDATE SET " +
    "    t.patrimonio = src.patrimonio, " +
    "    t.estado = src.estado, " +
    "    t.fecha_mod = src.fecha_mod " +
    "WHEN NOT MATCHED THEN " +
    "  INSERT (fec_periodo, patrimonio, estado, fecha_mod) " +
    "  VALUES (src.fec_periodo, src.patrimonio, src.estado, src.fecha_mod)";
