String mergeSql =
    "MERGE INTO DEHTGEMO.lineas_comprometidas t " +
    "USING (SELECT ? AS fec_periodo, ? AS contrato, ? AS monto, ? AS fecha_vcto FROM dual) src " +
    "ON (t.fec_periodo = src.fec_periodo AND t.contrato = src.contrato) " +
    "WHEN MATCHED THEN " +
    "  UPDATE SET " +
    "    t.monto = src.monto, " +
    "    t.fecha_vcto = src.fecha_vcto " +
    "WHEN NOT MATCHED THEN " +
    "  INSERT (fec_periodo, contrato, monto, fecha_vcto) " +
    "  VALUES (src.fec_periodo, src.contrato, src.monto, src.fecha_vcto)";
